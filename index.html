function fetchSheet(){

  var p2=fetch(apiUrl+'?action=getProducts&t='+t).then(function(r){return r.json();});

  Promise.all([p1,p2]).then(function(res){

    var rawT=res[0], rawP=res[1];

    if(rawT&&rawT.length>1){

      tasks=rawT.slice(1).map(function(r){

    if(rawP&&rawP.length>1){

    for(var j=0;j<STAGES.length;j++) opts+='<option value="'+STAGES[j]+'"'+(STAGES[j]===p.status?' selected':'')+'>'+SL[STAGES[j]]+'</option>';

    html+='<tr><td>'+(i+1)+'</td><td style="font-weight:700">'+esc(p.name)+'</td><td>'+p.pic+'</td>'

      +'<td>'+(p.checkout||'—')+'</td><td>'+(p.arrive||'—')+'</td>'

      +'<td><span class="sb '+p.status+'">'+(SL[p.status]||p.status)+'</span></td>'

      +'<td><div class="sdots">'+dots+'</div></td>'

      +'<td style="white-space:nowrap">'

      +'<select style="font-size:10px;padding:3px 5px;border:1px solid var(--border);border-radius:5px;background:var(--bg);font-family:inherit" onchange="updP('+i+',this.value)">'+opts+'</select>'

      +' <button class="btn sm danger" onclick="delP('+i+')">✕</button>'

      +'</td></tr>';

  }

  tb.innerHTML=html;

}


function updP(i,s){

  products[i].status=s;

  save('aff_prods',JSON.stringify(products));

  renderProducts();updateStats();pushSheet();showBanner('✅ Status diupdate');

}

}

function renderReport(){

  var cards='',rows='';

  for(var i=0;i<DIVS.length;i++){

    var d=DIVS[i];

    var all=tasks.filter(function(t){return t.div===d.id;});

    var dn=all.filter(function(t){return t.done;}).length;

    var pct=all.length?Math.round(dn/all.length*100):0;

    cards+='<div class="rc"><div class="rc-h '+d.id+'"><div class="rn">'+d.name+'</div><div class="rr">'+d.role+'</div></div>'

      +'<div class="rc-b">'

      +'<div class="rs"><span>Total</span><span class="rv">'+all.length+'</span></div>'

      +'<div class="rs"><span>Selesai</span><span class="rv" style="color:var(--green)">'+dn+'</span></div>'

      +'<div class="rs"><span>Belum</span><span class="rv" style="color:var(--red)">'+(all.length-dn)+'</span></div>'

      +'<div class="rs"><span>%</span><span class="rv">'+pct+'%</span></div>'

      +'</div></div>';

    rows+='<tr><td><b style="color:var(--'+d.id+')">'+d.name+'</b></td><td>'+all.length+'</td>'

      +'<td style="color:var(--green);font-weight:700">'+dn+'</td>'

      +'<td style="color:var(--red)">'+(all.length-dn)+'</td>'

      +'<td><div style="display:flex;align-items:center;gap:5px">'

      +'<div style="flex:1;height:4px;background:var(--border);border-radius:4px;overflow:hidden">'

      +'<div style="width:'+pct+'%;height:100%;background:var(--'+d.id+');border-radius:4px"></div></div>'

      +'<b style="font-size:10px">'+pct+'%</b></div></td></tr>';

  }

  document.getElementById('rep-cards').innerHTML=cards;

  document.getElementById('rep-rows').innerHTML=rows;

}

  for(var i=0;i<tasks.length;i++) if(tasks[i].id>=nid) nid=tasks[i].id+1;

  tasks.push({id:nid,div:document.getElementById('tm-div').value,day:+document.getElementById('tm-day').value,text:txt,tags:selTags.slice(),done:false});

  save('aff_tasks',JSON.stringify(tasks));

  cm('tm');updateStats();buildCal();pushSheet();showBanner('✅ Tugas ditambahkan!');

}
